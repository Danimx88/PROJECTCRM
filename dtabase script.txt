CREATE TABLE `agnordb`.`clientes` (
  `idcliente` INT NOT NULL,
  `viacontacto` VARCHAR(80) NOT NULL,
  `estatuscliente` VARCHAR(80) NOT NULL,
  `fechacontacto` DATE NOT NULL,
  `edadcliente` INT NOT NULL DEFAULT 18,
  `nombrecliente` VARCHAR(90) NOT NULL,
  `apellidoscliente` VARCHAR(100) NOT NULL,
  `correocliente` VARCHAR(80) NOT NULL,
  `clientescol` VARCHAR(45) NOT NULL,
  `telefonocel` INT NOT NULL,
  `telefonocasa` INT NOT NULL,
  `mediopublic` VARCHAR(90) NOT NULL,
  `detallemedio` VARCHAR(100) NOT NULL,
  `formapago` VARCHAR(100) NOT NULL,
  `montopago` DECIMAL(16,2) NOT NULL,
  `lineanegocio` VARCHAR(80) NOT NULL,
  `tipopropiedad` VARCHAR(80) NOT NULL,
  `rangoprecios` VARCHAR(80) NOT NULL,
  `notas` VARCHAR(150) NOT NULL,
  PRIMARY KEY (`idcliente`));
